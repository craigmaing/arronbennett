export default defineConfig({
  site: 'https://arronbennettbuilding.co.uk',
  integrations: [
    sitemap({
      filter: (page) => !page.includes('/admin/'),
      changefreq: 'weekly',
      priority: 0.7,
      lastmod: new Date()
    })

  image: {
    service: {
      entrypoint: 'astro/assets/services/sharp'

  build: {
    inlineStylesheets: 'auto'
  },
  compressHTML: true,
  experimental: {
    clientPrerender: true
  },
  vite: {
    plugins: [tailwindcss()]
  }
